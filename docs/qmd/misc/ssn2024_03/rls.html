tbl_rls <- 
tbl_r6xx |>
  filter(umur >= 25) |>
  mutate(
    konversi_ijazah_akhir = 
      case_when(
        ijazah_akhir == 25 ~ 0,
        ijazah_akhir |> between(1, 5) ~ 6,
        ijazah_akhir |> between(6, 10) ~ 9,
        ijazah_akhir |> between(11, 17) ~ 12,
        ijazah_akhir == 18 ~ 14, 
        ijazah_akhir == 19 ~ 15,
        ijazah_akhir |> between(20, 21) ~ 16,
        ijazah_akhir |> between(22, 24) ~ 18,
        .default = NA
      )
  ) |>
  mutate(
    lama_sekolah = case_when(
      partisipasi_sekolah == 1 ~ 0,
      
      partisipasi_sekolah == 2 & konversi_ijazah_akhir < 12  ~ konversi_ijazah_akhir + kelas_akhir - 1,
      partisipasi_sekolah == 2 & konversi_ijazah_akhir |> between(12, 15) & kelas_akhir < 8 ~ konversi_ijazah_akhir + 1,
      partisipasi_sekolah == 2 & konversi_ijazah_akhir >= 16 ~ konversi_ijazah_akhir + 1,

      
      
      partisipasi_sekolah == 3 & kelas_akhir < 8 & konversi_ijazah_akhir < 16 ~ konversi_ijazah_akhir + kelas_akhir - 1,
      partisipasi_sekolah == 3 & kelas_akhir < 8 & konversi_ijazah_akhir >= 16 ~ konversi_ijazah_akhir + 1,
      partisipasi_sekolah == 3 & kelas_akhir == 8 ~ konversi_ijazah_akhir,
      .default = NA
    )
  )

RLS menggunakan estimasi paket srvyr

library(srvyr)

srvyr_design_rls <- 
as_survey_design(
  .data = tbl_rls, 
  weights = FWT, 
  ids = c(PSU, SSU),
  strata = STRATA,
  nest = TRUE
)


srvyr_design_rls |>
  summarise(RLS = survey_mean(lama_sekolah))

RLS menggunakan paket survey

library(survey)

survey_design_rls <- svydesign(
  id = ~ 1, 
  data = tbl_rls, 
  weights = tbl_rls$FWT, 
  strata = tbl_rls$STRATA, 
  nest = TRUE
  )

svymean(x = ~ lama_sekolah, design = survey_design_rls)
               mean     SE
lama_sekolah 6.8042 0.1528

Kesimpulan

Jadi, kenapa angkanya masih berbeda ?

Kemungkinan ini karena profesional adjustment.

BPS melakukan sedikit perubahan sesuai dengan fenomena/kebijakan dari suatu daerah. RLS adalah indikator yang diestimasi secara top-down.
